<rbcloud-text-move-ani-group>

An abstract Animation Text Block component.

Example Usage

<template>
  <div class="container">

    <schedule-TextMoveAnimationGroup
      :active="data.active"
      :width="Number(data.width)"
      :delay="Number(data.delay)"
      :text-array="textArrays"
      :class-arr="data.classArr"
    />
  </div>
  <aside>

    <label>Display Text<input
      v-model="data.text"
      type="text"
    ></label>

    <label>Display Text<input
      v-model.number="data.width"
      type="range"
      min="100"
      max="1000"
    > {{ data.width }} px</label>
  </aside>
</template>

<script setup>

import { ref, computed, onMounted } from 'vue'

const data = ref({
  active: true,
  width: 1000,
  delay: 1,
  text: '테스트 텍스트 테스트 텍스트 테스트 텍스트 테스트 텍스트 ',
  classArr: []
})

const textArrays = computed(() => {
  return [data.value.text]
})

onMounted(() => {
})

</script>

Props

  • active : boolean
    블록 식벽 키.
    Defaults to false
  • width : string
    블록 식벽 키.
  • delay : number
    블록 식벽 키.
    Defaults to () => 0
  • textArray : array
    블록 식벽 키.
    Defaults to []
  • classArr : array
    블록 식벽 키.
    Defaults to []
