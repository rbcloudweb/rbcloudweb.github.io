<script setup>

import { ref, computed, onMounted } from 'vue'

import image1 from '@/assets/image/woman-7109043_1280.jpg'
import image2 from '@/assets/image/woman-8029209_1280.jpg'

const data = ref({
  menu_id: 'id_01',
  menu_name: '컨테이너 1',
  blockRows: [{
    epsd_id: '1',
    title: '',
    poster: image1
  }, {
    epsd_id: '2',
    title: '',
    poster: image2
  }],
  block: {
    height: 720,
    width: 480
  },
  container: {
    row: 1,
    position: {
      x: 100,
      y: 100
    },
    deg: -20,
    perspective: 1300,
    block_count: 30
  }
})

const displayBlockRows = ref([])

const selectedBlockContainerIdx = ref(0)
const selectedBlockContainerId = computed(() => {
  return selectedBlockContainerIdx.value + 1
  // return data.value.blockRows[selectedBlockContainerIdx.value]?.id || ''
})

function init() {
  console.log('rbcloud-block-container-example init')

  displayBlockRows.value = []
  const blocks = data.value.blockRows
  let index = 0
  let count = 0

  selectedItem(blocks[index])

  while (count < data.value.container.block_count) {
    const block = Object.assign({ epsd_id: `epsd_id_${count}` }, blocks[index])
    displayBlockRows.value.push(block)
    count += 1
    index += 1
    if (index === blocks.length) {
      index = 0
    }
  }
}

async function selectedItem(value) {
  // episode.value = value;
  console.log('episode selected callback', value)
}

onMounted(() => {
  init()
})

</script>

Props

  • title : string
    컨테이너 명.
    Defaults to ''
  • blockContainerId : string
    컨테이너 아이디.
    Defaults to ''
  • blockContainerIdx : number
    컨테이너 순번.
  • blockContainerPerspective : number
    컨테이너 원근감.
  • blockContainerPositionX : number
    컨테이너 위치 X.
  • blockContainerPositionY : number
    컨테이너 위치 Y.
  • blockContainerDeg : number
    컨테이너 위치 각도?.
  • blockList : array
    컨테이너 의 포스터 목록
  • isHorizontal : boolean
    컨테이너 의 포스터 가로 세로 구분
    Defaults to () => { return true }
  • selectedBlockContainerId : string
    컨테이너 의 선택된 컨테이너 아이디
  • selectedBlockContainerIdx : number
    컨테이너 의 선택된 컨테이너 순번
  • width : number
    포스터 넓이
  • height : number
    포스터 높이
